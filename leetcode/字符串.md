## 反转字符串
https://leetcode.cn/problems/reverse-string/description/

```java
// 双指针，前后开始
public void reverseString(char[] s) {
    if(s==null || s.length <2) return ;

    int left = 0, right = s.length-1;

    while(left<right){
        char tmp = s[left];
        s[left] = s[right];
        s[right] = tmp;
        left++;
        right--;
    }
    return ;
}
O(n) O(1)

// 递归
public void reverseString(char[] s) {
    if(s==null || s.length <2) return ;
    reverse(s,0,s.length-1);
    return ;
}

// 每次交换首尾两个元素，然后递归调用
public void reverse(char[] s,int start,int end){
    if(end <= start) return s;
    
    char tmp = s[start];
    s[start] = s[end];
    s[end] = tmp;
    
    reverse(s,start+1,end-1);
}
O(n) O(n)
```

## 反转字符串II
https://leetcode.cn/problems/reverse-string-ii/

```java
// 如果采用双指针，很复杂，要将字符串分段，每个分段进行要区分是逆序还是正序。
// 如果我们采用 stack 和 队列 会好很多。同时一般遍历就可以
public String reverseStr(String s, int k) {
    if(s==null) return null;

    StringBuilder sb = new StringBuilder();
    Stack<Character> stack = new Stack();
    boolean reverse = true;
    int count = k;
    for(int i = 0 ;i<s.length() ; i++){
        if(reverse){
            stack.push(s.charAt(i));
        }else{
            sb.append(s.charAt(i));
        }
        if(--count == 0){
           if(reverse){
                while(!stack.isEmpty()){
                    sb.append(stack.pop());
                }
           }
           reverse=!reverse;
           count = k;
        }
    }

    //如果有将stcak清空
    while(!stack.isEmpty()){
        sb.append(stack.pop());
    }

    return sb.toString();

}

public String reverseStr(String s, int k) {
    if(s==null) return null;
    char[] cs = s.toCharArray();

    for(int i=0;i<s.length();i+=2k){
```

## 反转字符串里的单词
https://leetcode.cn/problems/reverse-words-in-a-string/description/

```java
// 使用额外空间来操作
// 因为是反转顺序，因此考虑用stack，但是如果用sb 进行清空操作 是不行的，因此其实还是对字符串整体进行拼接和操作
// 综上 直接使用官方思路：1.清空额外字符串 2.整体进行反转 
public String reverseWords(String s) {
    StringBuilder sb = trimSpaces(s);

    reverse(sb,0,sb.length()-1);

    reverseWord(sb);

    return sb.toString();
}

public void reverseWord(StringBuilder sb){
    int slow =0, fast =0;
    while(fast<sb.length()){
        if(sb.charAt(fast)==' '){
            reverse(sb,slow,fast-1);
            slow = ++fast;
        }else
            ++fast;
    }
    reverse(sb,slow,sb.length()-1)

}

public void reverse(StringBuilder sb, int start, int end){
    while(start<end){
        char tmp = sb.charAt(start);
        sb.setCharAt(start,sb.charAt(end));
        sb.setCharAt(end,tmp);
        start++;
        end--;
    }
}

public StringBuilder trimSpaces(String s){
    StringBuilder sb = new StringBuilder();
    boolean word = false;
    for(int i=0; i<s.length();i++){
        char c = s.charAt(i);
        if(c!=' '){
            word = true;
            sb.append(c);
        }else if(word == true){
            word = false;
            sb.append(c);
        }else{
            // do Nothing;
        }
    }
    if(sb.charAt(sb.length()-1)==' ')
        sb.deleteCharAt(sb.length()-1);
    return sb;
}
```
